program patito;
int a[5];
{
    def void main()
    int i, k, temp;
    {
        for(i = 0; i < 5; i = i + 1) {
            input(k);
            a[i] = k;
        }

        for(i = 0; i < 5; i = i + 1) {
            for(k = 0; k < 5 - 1; k = k + 1){
                if(a[k] > a[k + 1]){
                    temp = a[k];
                    a[k] = a[k + 1];
                    a[k + 1] = temp;
                }
            }
        }
        for(i = 0; i < 5; i = i + 1) {
            print(a[i]);
        }
    }
}
